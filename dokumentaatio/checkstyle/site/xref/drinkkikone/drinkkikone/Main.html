16      public static void main(String[] args) {
17          //Perjantain demoa varten Mainiin lisätty pientä testidataa
18          Baarikaappi baarikaappi = new Baarikaappi();
19          Reseptikirja reseptikirja = new Reseptikirja();
20          Scanner lukija = new Scanner(System.in);
21          
22          Ainesosa vermutti = new Ainesosa("Vermutti", 15.0);
23          Ainesosa gin = new Ainesosa("Gin", 40.0);
24          Ainesosa tonic = new Ainesosa("Tonic -vesi", 0.0); 
25          Ainesosa rommi = new Ainesosa("Rommi", 35.0);
26          Ainesosa kola = new Ainesosa("Coca-cola", 0.0);
27          
28          baarikaappi.lisaaAinesosa(gin, 5.0);
29          baarikaappi.lisaaAinesosa(tonic, 20.0);
30          baarikaappi.lisaaAinesosa(vermutti, 1.0);
31          baarikaappi.lisaaAinesosa(rommi, 100.0);
32          baarikaappi.lisaaAinesosa(kola, 100.0);
33          
34          HashMap<Ainesosa, Double> rommikolanAinesosat = new HashMap<>();
35          rommikolanAinesosat.put(rommi, 2.0);
36          rommikolanAinesosat.put(kola, 2.0);
37          
38          Resepti resepti1 = new Resepti("Martini");
39          resepti1.lisaaAinesosa(gin, 6.0);
40          resepti1.lisaaAinesosa(vermutti, 1.0);
41          resepti1.setValmistusohje("Laita ravistimeen jäitä, lisää ainekset ja hämmennä.");
42          
43          Resepti resepti2 = new Resepti("Gin & Tonic");
44          resepti2.lisaaAinesosa(gin, 4.0);
45          resepti2.lisaaAinesosa(tonic, 12.0);
46          resepti2.setValmistusohje("Täytä lasi puolilleen jäitä, lisää gini ja pidennä tonic-vedellä.");
47          
48          Resepti resepti3 = new Resepti("Rommikola");
49          resepti3.lisaaUseaAinesosa(rommikolanAinesosat);
50          resepti3.setValmistusohje("Täytä highball-lasi jääpaloilla. Kaada lasiin ensin rommi ja sitten cola. Sekoita baarilusikalla.");
51          
52          Drinkkikone drinkkikone = new Drinkkikone(baarikaappi, reseptikirja);
53          drinkkikone.lisaaResepti(resepti1);
54          drinkkikone.lisaaResepti(resepti2);
55          drinkkikone.lisaaResepti(resepti3);
56          
57          TekstiKayttoliittyma tKayttoliittyma = new TekstiKayttoliittyma(drinkkikone, lukija);
58          GraafinenKayttoliittyma gKayttoliittyma = new GraafinenKayttoliittyma(drinkkikone);
59          
60          tKayttoliittyma.suorita();
61          gKayttoliittyma.run();
62      }
